    function calculate() {

      const baseAmount = parseFloat(document.getElementById('amount').value);

      const months = parseInt(document.getElementById('months').value);

      const withDP = document.getElementById('withDownPayment').checked;


      if (isNaN(baseAmount) || baseAmount <= 0) {

        document.getElementById('result').innerHTML = "Введите корректную сумму.";

        return;

      }


      // Надбавка по диапазонам

        surcharge = 4000;

      } else {

        surcharge = 5000;

      }


      const adjustedAmount = baseAmount + surcharge;


      const interest = withDP ? interestWithDown[months] : interestWithoutDown[months];

      const total = adjustedAmount * (1 + interest / 100);

      const downPayment = withDP ? total * 0.25 : 0;

      const monthly = (total - downPayment) / months;
